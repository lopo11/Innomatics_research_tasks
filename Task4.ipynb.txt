        "    if len(password) < 8:\n",
        "        return \"Weak Password\"\n",

        "    has_digit = False\n",
        "    has_special = False\n",
        "\n",
        "    for ch in password:\n",
        "        if ch.isdigit():\n",
        "            has_digit = True\n",
        "\n",
        "        if ch in \"@#$\":\n",
        "            has_special = True\n",
        "\n",
        "    if has_digit and has_special:\n",
        "        return \"Strong Password\"\n",
        "    else:\n",
        "        return \"Weak Password\"\n",
        "\n",
